attribute vec4 vertex;
attribute vec4 color;
attribute vec2 texcoord;

uniform Matrix4x4 ObjectToWorld;
uniform Matrix4x4 MATRIX_VP;

varying vec4 color;
varying vec2 texcoord;
varying vec3 worldPos;

main(){
	worldPos = mul(ObjectToWorld, vertex);
	gl_Position = Global_ObjectToClipPos(vertex);
	texcoord = texcoord;
	color = color;
	return;//gl_Position is set

}
vec4 Global_ObjectToClipPos(vec4 inVertex)
{
	return mul(MATRIX_VP, mul(ObjectToWorld, inVertex));

}
